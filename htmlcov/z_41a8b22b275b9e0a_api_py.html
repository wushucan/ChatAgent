55 def send_message(self, session_id: str, message: str, web_search: bool = False, context_compression: bool = True) -> dict: 

56 """Start streaming a reply in a background thread, return immediately.""" 

57 if not session_id: 

58 return {"status": "error", "error": "无效的会话 ID"} 

59 # Persist user message 

60 self._store.add_message(session_id, "user", message) 

61 

62 # Prepare stream buffer (includes interrupt support) 

63 buffer: dict[str, Any] = { 

64 "tokens": [], 

65 "sent_count": 0, 

66 "done": False, 

67 "error": None, 

68 "searching": False, 

69 "interrupt": None, # interrupt data when waiting for user 

70 "resume_event": threading.Event(), # signal to resume after interrupt 

71 "resume_value": {}, # user's confirmation response 

72 } 

73 self._buffers[session_id] = buffer 

74 

75 # No history loading — checkpoint handles accumulation 

76 history = self._store.get_messages(session_id) 

77 

78 def _run(): 

79 gen = stream_response( 

80 history, 

81 self._config, 

82 web_search=web_search, 

83 thread_id=session_id, 

84 context_compression=context_compression, 

85 rag_engine=self._rag, 

86 memory_store=self._memory, 

87 ) 

88 resume_value = None 

89 try: 

90 while True: 

91 try: 

92 if resume_value is not None: 

93 typ, val = gen.send(resume_value) 

94 resume_value = None 

95 else: 

96 typ, val = next(gen) 

97 except StopIteration: 

98 break 

99 

100 if typ == "interrupt": 

101 # Graph paused — wait for user to confirm via confirm_tool() 

102 buffer["interrupt"] = val 

103 buffer["resume_event"].wait() 

104 buffer["resume_event"].clear() 

105 buffer["interrupt"] = None 

106 resume_value = buffer.pop("resume_value", {}) 

107 continue 

108 

109 if typ == "token": 

110 buffer["tokens"].append(val) 

111 elif typ == "search": 

112 buffer["searching"] = val if isinstance(val, str) else bool(val) 

113 

114 # Regenerate session title based on full conversation 

115 full = "".join(buffer["tokens"]) 

116 title = generate_title(history + [{"role": "assistant", "content": full}], self._config) 

117 if title: 

118 self._store.rename_session(session_id, title) 

119 

120 except Exception as e: 

121 buffer["error"] = str(e) 

122 finally: 

123 buffer["done"] = True 

124 

125 t = threading.Thread(target=_run, daemon=True) 

126 t.start() 

127 return {"status": "started"}

129 def get_stream_output(self, session_id: str) -> dict: 

130 """Return accumulated tokens since last poll (called from JS).""" 

131 buffer = self._buffers.get(session_id) 

132 if not buffer: 

133 return {"tokens": [], "done": True} 

134 

135 new_tokens = buffer["tokens"][buffer["sent_count"] :] 

136 buffer["sent_count"] = len(buffer["tokens"]) 

137 

138 searching = buffer.get("searching", False) 

139 done = buffer["done"] 

140 result: dict[str, Any] = { 

141 "tokens": new_tokens, 

142 "done": done, 

143 "searching": searching, 

144 } 

145 

146 # Propagate interrupt data to frontend 

147 if buffer.get("interrupt"): 

148 result["interrupt"] = buffer["interrupt"] 

149 

150 if buffer.get("error") and done: 

151 result["error"] = buffer["error"] 

152 

153 # When stream ends, persist the assistant message and clean up 

154 if done: 

155 full_text = "".join(buffer["tokens"]) 

156 self._store.add_message(session_id, "assistant", full_text) 

157 self._buffers.pop(session_id, None) 

158 

159 return result

163 def confirm_tool(self, session_id: str, approved: bool = True) -> dict: 

164 """Resume the graph after an interrupt with user's decision. 

165 

166 Called from JS when the user confirms or cancels a tool call. 

167 """ 

168 buffer = self._buffers.get(session_id) 

169 if not buffer or buffer.get("interrupt") is None: 

170 return {"status": "no_interrupt"} 

171 

172 buffer["resume_value"] = "approved" if approved else "rejected" 

173 buffer["resume_event"].set() 

174 return {"status": "resumed"}
